# Binary Wire Protocol via Protobuf — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace JSON wire encoding (WS + HTTP) with Protocol Buffers; schema lives in `backend/crates/protocol/proto/abutown.proto` as the single source of truth.

**Architecture:** Schema-first. Rust codegen via `prost-build` in `build.rs`. TS codegen via `@bufbuild/protoc-gen-es`. DB persistence stays JSON (separate storage DTOs). Atomic cutover — no JSON fallback on the wire.

**Tech Stack:** `prost = "0.13"`, `prost-build = "0.13"`, `@bufbuild/protobuf` (TS runtime), `@bufbuild/buf` + `@bufbuild/protoc-gen-es` (TS codegen). System `protoc` binary required (libprotoc 3.20+).

**Spec:** `docs/superpowers/specs/2026-05-20-binary-wire-protobuf-design.md`.

import { spawnSync } from 'node:child_process';
import { mkdirSync, existsSync } from 'node:fs';

const here = dirname(fileURLToPath(import.meta.url));
const root = resolve(here, '..');

mkdirSync(outDir, { recursive: true });

  stdio: 'inherit',

use abutown_protocol::v1::*;
use criterion::{Criterion, criterion_group, criterion_main};
use prost::Message;

fn sample_delta(n_agents: usize) -> ServerMessage {
    let agents: Vec<AgentMobility> = (0..n_agents).map(|i| AgentMobility {
        id: format!("agent:walker:{i}"),
        state: Some(AgentState {
            state: Some(agent_state::State::Walking(Walking {
                link_id: "link:walk:corridor:22".into(),
                progress: (i as f32) / (n_agents as f32),
            })),
        }),
        world_coord: Some(WorldCoord { x: 92.0 + i as f32, y: 133.0 }),
        direction: Direction::E as i32,
        sprite_key: format!("pedestrian:{}", i % 16),
        plan_cursor: 0,
    }).collect();

    ServerMessage {
        body: Some(server_message::Body::MobilityChunkDelta(MobilityChunkDelta {
            protocol_version: 16,
            world_id: "abutown-main".into(),
            tick: 1234,
            chunk: Some(ChunkCoord { x: 4, y: 4 }),
            changed_agents: agents,
            changed_vehicles: vec![],
            left_agents: vec![],
            left_vehicles: vec![],
        })),
    }
}

fn json_equivalent_for_delta(n_agents: usize) -> String {
    // Build the same payload as JSON via serde_json::Value for an apples-to-apples
    // byte-count comparison.
    let agents: Vec<serde_json::Value> = (0..n_agents).map(|i| serde_json::json!({
        "id": format!("agent:walker:{i}"),
        "state": { "type": "walking", "link_id": "link:walk:corridor:22", "progress": (i as f32) / (n_agents as f32) },
        "world_coord": { "x": 92.0 + i as f32, "y": 133.0 },
        "direction": "e",
        "sprite_key": format!("pedestrian:{}", i % 16),
        "plan_cursor": 0,
    })).collect();
    let v = serde_json::json!({
        "type": "mobility_chunk_delta",
        "protocol_version": 16,
        "world_id": "abutown-main",
        "tick": 1234,
        "chunk": { "x": 4, "y": 4 },
        "changed_agents": agents,
        "changed_vehicles": [],
        "left_agents": [],
        "left_vehicles": [],
    });
    serde_json::to_string(&v).unwrap()
}

fn bench_wire_size(c: &mut Criterion) {
    for &n in &[1usize, 10, 50, 100] {
        let proto = sample_delta(n);
        let proto_bytes = proto.encode_to_vec();
        let json_str = json_equivalent_for_delta(n);
        println!(
            "{n:>4} agents | JSON {:>7} bytes | Proto {:>5} bytes | ratio {:.2}x",
            json_str.len(),
            proto_bytes.len(),
            json_str.len() as f64 / proto_bytes.len() as f64,
        );
    }

    c.bench_function("encode_50_agent_delta", |b| {
        let msg = sample_delta(50);
        b.iter(|| {
            let _ = msg.encode_to_vec();
        });
    });
}

criterion_group!(benches, bench_wire_size);
criterion_main!(benches);
